8 function dhtmlXTreeFromHTML(obj){

17  if (obj.childNodes[j].nodeType=="1"){
18  if (obj.childNodes[j].tagName=="XMP"){
19  var cHead=obj.childNodes[j];
20  for (var m=0; m<cHead.childNodes.length; m++)
21  cont+=cHead.childNodes[m].data;
22 
23  }
24  else if (obj.childNodes[j].tagName.toLowerCase()=="ul")
25  cont=dhx_li2trees(obj.childNodes[j],new Array(),0);

71 function dhx_li2trees(tag,data,ind){
72  for (var i=0; i<tag.childNodes.length; i++){
73  var z=tag.childNodes[i];
74  if ((z.nodeType==1)&&(z.tagName.toLowerCase()=="li")){
75  var c=""; var ul=null;
76  var check=z.getAttribute("checked");
77  for (var j=0; j<z.childNodes.length; j++){
78  var zc=z.childNodes[j];
79  if (zc.nodeType==3) c+=zc.data;
80  else if (zc.tagName.toLowerCase()!="ul") c+=dhx_outer_html(zc);
81  else ul=zc;
82  }
83 
84  data[data.length]=[ind,c,check,(z.id||(data.length+1))];
85  if (ul)
86  data=dhx_li2trees(ul,data,(z.id||data.length));
87  }
88  }
89  return data;
90 }
