36 document.getElemetsByClassName = function(name,target)
37 {
38  var result = [];
39  var object = null;
40  var search = new RegExp(['(^|\\s)',name,'(\\s|$)'].join(''));
41  if (target && target.getElementsByTagName)
42  object = target.getElementsByTagName('*');
43  if (!object)
44  object = document.getElementsByTagName ? document.getElementsByTagName('*') : document.all;

46  {
